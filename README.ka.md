## ­Ъје рЃцрЃБрЃюрЃЦрЃфрЃўрЃЮрЃюрЃљрЃџрЃў

РюЁ **рЃЌрЃЋрЃћрЃћрЃЉрЃќрЃћ рЃюрЃљрЃЋрЃўрЃњрЃљрЃфрЃўрЃљ** - Рєљ Рєњ рЃдрЃўрЃџрЃљрЃЎрЃћрЃЉрЃўрЃЌ рЃгрЃўрЃюрЃљ/рЃерЃћрЃЏрЃЊрЃћрЃњ рЃЌрЃЋрЃћрЃќрЃћ рЃњрЃљрЃЊрЃљрЃАрЃЋрЃџрЃљ  
РюЁ **рЃЊрЃдрЃћрЃЋрЃљрЃюрЃЊрЃћрЃџрЃў рЃЊрЃдрЃўрЃА рЃЏрЃЮрЃюрЃўрЃерЃЋрЃюрЃљ** - рЃџрЃБрЃарЃ»рЃў border рЃЊрЃдрЃћрЃЋрЃљрЃюрЃЊрЃћрЃџ рЃЌрЃљрЃарЃўрЃдрЃќрЃћ  
РюЁ **рЃљрЃарЃЕрЃћрЃБрЃџрЃў рЃЊрЃдрЃўрЃА highlighting** - рЃџрЃБрЃарЃ»рЃў background рЃљрЃарЃЕрЃћрЃБрЃџ рЃЌрЃљрЃарЃўрЃдрЃќрЃћ  
РюЁ **Click Outside** - рЃЎрЃљрЃџрЃћрЃюрЃЊрЃљрЃарЃўрЃА рЃњрЃљрЃарЃћрЃЌ рЃЊрЃљрЃЎрЃџрЃўрЃЎрЃћрЃЉрЃўрЃАрЃљрЃА popup рЃўрЃ«рЃБрЃарЃћрЃЉрЃљ  
РюЁ **Responsive Design** - рЃЏрЃЮрЃарЃњрЃћрЃЉрЃБрЃџрЃў 7x6 grid layout  
РюЁ **рЃЎрЃЋрЃўрЃарЃўрЃА рЃЊрЃдрЃћрЃћрЃЉрЃў** - Sun, Mon, Tue, Wed, Thu, Fri, Sat  
РюЁ **рЃЦрЃљрЃарЃЌрЃБрЃџрЃў рЃЌрЃЋрЃћрЃћрЃЉрЃўрЃА рЃЏрЃ«рЃљрЃарЃЊрЃљрЃГрЃћрЃарЃљ** - рЃљрЃЊрЃЋрЃўрЃџрЃљрЃЊ рЃерЃћрЃўрЃфрЃЋрЃџрЃћрЃЉрЃљ `getMonthName` рЃцрЃБрЃюрЃЦрЃфрЃўрЃљрЃерЃў

## ­ЪДа рЃЎрЃЮрЃЏрЃърЃЮрЃюрЃћрЃюрЃбрЃўрЃА рЃљрЃарЃЦрЃўрЃбрЃћрЃЦрЃбрЃБрЃарЃљ

### **State Management (3 State)**

```tsx
const [selectedDate, setSelectedDate] = useState<Date | null>(null);
const [currentMonth, setCurrentMonth] = useState<Date>(new Date());
const [isOpen, setIsOpen] = useState(false);
```

1. **`selectedDate`** - рЃарЃЮрЃЏрЃћрЃџрЃў рЃЌрЃљрЃарЃўрЃдрЃўрЃљ рЃљрЃарЃЕрЃћрЃБрЃџрЃў рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃџрЃўрЃА рЃЏрЃўрЃћрЃа
2. **`currentMonth`** - рЃарЃЮрЃЏрЃћрЃџ рЃЌрЃЋрЃћрЃА рЃљрЃЌрЃЋрЃљрЃџрЃўрЃћрЃарЃћрЃЉрЃА рЃљрЃ«рЃџрЃљ (рЃерЃћрЃўрЃФрЃџрЃћрЃЉрЃљ рЃњрЃљрЃюрЃАрЃ«рЃЋрЃљрЃЋрЃЊрЃћрЃЉрЃЮрЃЊрЃћрЃА рЃљрЃарЃЕрЃћрЃБрЃџрЃўрЃАрЃњрЃљрЃю)
3. **`isOpen`** - рЃЎрЃљрЃџрЃћрЃюрЃЊрЃљрЃарЃўрЃА popup рЃдрЃўрЃљрЃљ рЃЌрЃБ рЃЊрЃљрЃ«рЃБрЃарЃБрЃџрЃў

### **Click Outside Detection**

```tsx
const calendarRef = useRef<HTMLDivElement>(null);

useEffect(() => {
  const handleClickOutside = (event: MouseEvent) => {
    if (calendarRef.current && !calendarRef.current.contains(event.target as Node)) {
      setIsOpen(false);
    }
  };
  document.addEventListener("mousedown", handleClickOutside);
  return () => document.removeEventListener("mousedown", handleClickOutside);
}, []);
```

**рЃарЃЮрЃњрЃЮрЃа рЃЏрЃБрЃерЃљрЃЮрЃЉрЃА:**
1. `useRef` - рЃЦрЃЏрЃюрЃўрЃА reference-рЃА рЃЎрЃљрЃџрЃћрЃюрЃЊрЃљрЃарЃўрЃА container-рЃќрЃћ
2. `useEffect` - document-рЃќрЃћ рЃљрЃЏрЃљрЃбрЃћрЃЉрЃА click listener-рЃА (mount-рЃќрЃћ)
3. `handleClickOutside` - рЃљрЃЏрЃЮрЃгрЃЏрЃћрЃЉрЃА click рЃўрЃДрЃЮ рЃЎрЃљрЃџрЃћрЃюрЃЊрЃљрЃарЃўрЃА рЃерЃўрЃњрЃюрЃўрЃЌ рЃЌрЃБ рЃњрЃљрЃарЃћрЃЌ
4. рЃЌрЃБ рЃњрЃљрЃарЃћрЃЌ Рєњ `setIsOpen(false)`
5. Cleanup рЃцрЃБрЃюрЃЦрЃфрЃўрЃљ - рЃгрЃљрЃерЃџрЃўрЃА listener-рЃА (unmount-рЃќрЃћ, memory leak-рЃўрЃА рЃЌрЃљрЃЋрЃўрЃЊрЃљрЃю рЃљрЃАрЃљрЃфрЃўрЃџрЃћрЃЉрЃџрЃљрЃЊ)

### **рЃЌрЃљрЃарЃўрЃдрЃўрЃА рЃцрЃЮрЃарЃЏрЃљрЃбрЃўрЃарЃћрЃЉрЃљ**

```tsx
const formatDate = (date: Date | null) => {
  if (!date) return "choose date";
  const day = date.getDate();
  const month = date.getMonth() + 1;  // +1 рЃарЃљрЃЊрЃњрЃљрЃю getMonth() рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃА 0-11
  const year = date.getFullYear();
  return `${day}/${month}/${year}`;
};
```

**JavaScript Date рЃЏрЃћрЃЌрЃЮрЃЊрЃћрЃЉрЃў:**
- `getDate()` Рєњ рЃЊрЃдрЃћ (1-31)
- `getMonth()` Рєњ рЃЌрЃЋрЃћ (0-11) Рџа№ИЈ рЃўрЃљрЃюрЃЋрЃљрЃарЃў = 0!
- `getFullYear()` Рєњ рЃгрЃћрЃџрЃў (2025)
- `getDay()` Рєњ рЃЎрЃЋрЃўрЃарЃўрЃА рЃЊрЃдрЃћ (0-6, рЃЎрЃЋрЃўрЃарЃљ = 0)

### **рЃЌрЃЋрЃўрЃА рЃАрЃљрЃ«рЃћрЃџрЃўрЃА рЃЏрЃўрЃдрЃћрЃЉрЃљ**

```tsx
const getMonthName = (date: Date): string => {
  const months = [
    "January", "February", "March", "April", "May", "June",
    "July", "August", "September", "October", "November", "December"
  ];
  return months[date.getMonth()];
};
```

**рЃЦрЃљрЃарЃЌрЃБрЃџрЃљрЃЊ рЃерЃћрЃАрЃљрЃфрЃЋрЃџрЃћрЃџрЃљрЃЊ:**
```tsx
const months = [
  "рЃўрЃљрЃюрЃЋрЃљрЃарЃў", "рЃЌрЃћрЃЉрЃћрЃарЃЋрЃљрЃџрЃў", "рЃЏрЃљрЃарЃбрЃў", "рЃљрЃърЃарЃўрЃџрЃў", "рЃЏрЃљрЃўрЃАрЃў", "рЃўрЃЋрЃюрЃўрЃАрЃў",
  "рЃўрЃЋрЃџрЃўрЃАрЃў", "рЃљрЃњрЃЋрЃўрЃАрЃбрЃЮ", "рЃАрЃћрЃЦрЃбрЃћрЃЏрЃЉрЃћрЃарЃў", "рЃЮрЃЦрЃбрЃЮрЃЏрЃЉрЃћрЃарЃў", "рЃюрЃЮрЃћрЃЏрЃЉрЃћрЃарЃў", "рЃЊрЃћрЃЎрЃћрЃЏрЃЉрЃћрЃарЃў"

```

### **рЃЌрЃЋрЃћрЃћрЃЉрЃќрЃћ рЃюрЃљрЃЋрЃўрЃњрЃљрЃфрЃўрЃљ**

```tsx
const goToPreviousMonth = () => {
  setCurrentMonth(
    new Date(currentMonth.getFullYear(), currentMonth.getMonth() - 1, 1)
  );
};

const goToNextMonth = () => {
  setCurrentMonth(
    new Date(currentMonth.getFullYear(), currentMonth.getMonth() + 1, 1)
  );
};
```

**JavaScript-рЃўрЃА рЃЏрЃљрЃњрЃўрЃљ:**
- `new Date(2025, -1, 1)` Рєњ 2024 рЃЊрЃћрЃЎрЃћрЃЏрЃЉрЃћрЃарЃў (рЃљрЃЋрЃбрЃЮрЃЏрЃљрЃбрЃБрЃарЃў рЃЎрЃЮрЃарЃћрЃЦрЃфрЃўрЃљ!)
- `new Date(2025, 12, 1)` Рєњ 2026 рЃўрЃљрЃюрЃЋрЃљрЃарЃў
- рЃЌрЃЋрЃћ -1 рЃљрЃю +1 Рєњ JavaScript рЃЌрЃљрЃЋрЃљрЃЊ рЃљрЃАрЃгрЃЮрЃарЃћрЃЉрЃА рЃгрЃћрЃџрЃА рЃЌрЃБ рЃАрЃљрЃГрЃўрЃарЃЮрЃљ

### **рЃЌрЃЋрЃўрЃА рЃЊрЃдрЃћрЃћрЃЉрЃўрЃА рЃарЃљрЃЮрЃЊрЃћрЃюрЃЮрЃЉрЃўрЃА рЃњрЃљрЃЏрЃЮрЃЌрЃЋрЃџрЃљ**

```tsx
const getDayInMonth = (date: Date): number => {
  return new Date(date.getFullYear(), date.getMonth() + 1, 0).getDate();
};
```

**рЃарЃЮрЃњрЃЮрЃа рЃЏрЃБрЃерЃљрЃЮрЃЉрЃА:**
- `new Date(2025, 12, 0)` = рЃўрЃљрЃюрЃЋрЃарЃўрЃА 0-рЃћ рЃЊрЃдрЃћ = **рЃЊрЃћрЃЎрЃћрЃЏрЃЉрЃарЃўрЃА рЃЉрЃЮрЃџрЃЮ рЃЊрЃдрЃћ** (31)
- `new Date(2025, 2, 0)` = рЃЏрЃљрЃарЃбрЃўрЃА 0-рЃћ рЃЊрЃдрЃћ = **рЃЌрЃћрЃЉрЃћрЃарЃЋрЃџрЃўрЃА рЃЉрЃЮрЃџрЃЮ рЃЊрЃдрЃћ** (28 рЃљрЃю 29)
- **рЃерЃћрЃЏрЃЊрЃћрЃњрЃў рЃЌрЃЋрЃўрЃА 0-рЃћ рЃЊрЃдрЃћ = рЃљрЃЏ рЃЌрЃЋрЃўрЃА рЃЉрЃЮрЃџрЃЮ рЃЊрЃдрЃћ!**

**рЃерЃћрЃЊрЃћрЃњрЃћрЃЉрЃў:**
- рЃўрЃљрЃюрЃЋрЃљрЃарЃў, рЃЏрЃљрЃарЃбрЃў, рЃЏрЃљрЃўрЃАрЃў, рЃўрЃЋрЃџрЃўрЃАрЃў, рЃљрЃњрЃЋрЃўрЃАрЃбрЃЮ, рЃЮрЃЦрЃбрЃЮрЃЏрЃЉрЃћрЃарЃў, рЃЊрЃћрЃЎрЃћрЃЏрЃЉрЃћрЃарЃў Рєњ 31
- рЃљрЃърЃарЃўрЃџрЃў, рЃўрЃЋрЃюрЃўрЃАрЃў, рЃАрЃћрЃЦрЃбрЃћрЃЏрЃЉрЃћрЃарЃў, рЃюрЃЮрЃћрЃЏрЃЉрЃћрЃарЃў Рєњ 30
- рЃЌрЃћрЃЉрЃћрЃарЃЋрЃљрЃџрЃў Рєњ 28 (рЃљрЃю 29 leap year-рЃерЃў)

**рЃарЃљрЃА рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃА:**
- 0 = рЃЎрЃЋрЃўрЃарЃљ (Sunday)
- 1 = рЃЮрЃарЃерЃљрЃЉрЃљрЃЌрЃў (Monday)
- 2 = рЃАрЃљрЃЏрЃерЃљрЃЉрЃљрЃЌрЃў (Tuesday)
- ...рЃЊрЃљ рЃљ.рЃе.

**рЃарЃљрЃбрЃЮрЃЏ рЃњрЃЋрЃГрЃўрЃарЃЊрЃћрЃЉрЃљ:**
- grid-рЃерЃў рЃарЃљрЃЏрЃЊрЃћрЃюрЃў рЃфрЃљрЃарЃўрЃћрЃџрЃў рЃБрЃ»рЃарЃљ рЃБрЃюрЃЊрЃљ рЃўрЃДрЃЮрЃА рЃЌрЃЋрЃўрЃА рЃЊрЃљрЃАрЃљрЃгрЃДрЃўрЃАрЃерЃў
- рЃЏрЃљрЃњ: рЃЌрЃБ рЃЌрЃЋрЃћ рЃўрЃгрЃДрЃћрЃЉрЃљ рЃ«рЃБрЃЌрЃерЃљрЃЉрЃљрЃЌрЃА (4) Рєњ 3 рЃфрЃљрЃарЃўрЃћрЃџрЃў рЃБрЃ»рЃарЃљ (рЃЮрЃарЃе, рЃАрЃљрЃЏ, рЃЮрЃЌрЃ«)

### **рЃЎрЃљрЃџрЃћрЃюрЃЊрЃљрЃарЃўрЃА рЃЊрЃдрЃћрЃћрЃЉрЃўрЃА рЃЏрЃљрЃАрЃўрЃЋрЃўрЃА рЃњрЃћрЃюрЃћрЃарЃљрЃфрЃўрЃљ**

```tsx
const generateCaledarDays = () => {
  const days: (number | null)[] = [];
  const dayInMonth = getDayInMonth(currentMonth);
  const firstDay = getFirstDayOfMonth(currentMonth);
  const emptyState = firstDay === 0 ? 6 : firstDay - 1;

  // рЃфрЃљрЃарЃўрЃћрЃџрЃў рЃБрЃ»рЃарЃћрЃЉрЃў
  for (let i = 0; i < emptyState; i++) {
    days.push(null);
  }
  
  // рЃЊрЃдрЃћрЃћрЃЉрЃў
  for (let day = 1; day <= dayInMonth; day++) {
    days.push(day);
  }

  return days;
};
```

**рЃарЃЮрЃњрЃЮрЃа рЃЏрЃБрЃерЃљрЃЮрЃЉрЃА:**

1. **рЃфрЃљрЃарЃўрЃћрЃџрЃў рЃБрЃ»рЃарЃћрЃЉрЃўрЃА рЃњрЃљрЃЏрЃЮрЃЌрЃЋрЃџрЃљ:**
   - рЃЌрЃБ рЃЌрЃЋрЃћ рЃўрЃгрЃДрЃћрЃЉрЃљ рЃЎрЃЋрЃўрЃарЃљрЃА (0) Рєњ 6 рЃфрЃљрЃарЃўрЃћрЃџрЃў (grid рЃўрЃгрЃДрЃћрЃЉрЃљ рЃЮрЃарЃерЃљрЃЉрЃљрЃЌрЃўрЃЊрЃљрЃю!)
   - рЃЌрЃБ рЃЌрЃЋрЃћ рЃўрЃгрЃДрЃћрЃЉрЃљ рЃЮрЃарЃерЃљрЃЉрЃљрЃЌрЃА (1) Рєњ 0 рЃфрЃљрЃарЃўрЃћрЃџрЃў
   - рЃАрЃ«рЃЋрЃљ: `firstDay - 1`

2. **рЃЏрЃљрЃАрЃўрЃЋрЃўрЃА рЃерЃћрЃЋрЃАрЃћрЃЉрЃљ:**
   - рЃ»рЃћрЃа null-рЃћрЃЉрЃў (рЃфрЃљрЃарЃўрЃћрЃџрЃў рЃБрЃ»рЃарЃћрЃЉрЃў)
   - рЃерЃћрЃЏрЃЊрЃћрЃњ рЃарЃўрЃфрЃ«рЃЋрЃћрЃЉрЃў 1-рЃЊрЃљрЃю 28/30/31-рЃЏрЃЊрЃћ

**рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў - рЃЊрЃћрЃЎрЃћрЃЏрЃЉрЃћрЃарЃў 2025:**
- рЃўрЃгрЃДрЃћрЃЉрЃљ рЃЮрЃарЃерЃљрЃЉрЃљрЃЌрЃА (1) Рєњ 0 рЃфрЃљрЃарЃўрЃћрЃџрЃў
- рЃљрЃЦрЃЋрЃА 31 рЃЊрЃдрЃћ
- рЃЏрЃљрЃАрЃўрЃЋрЃў: `[1, 2, 3, 4, 5, 6, 7, 8, ..., 30, 31]`

**рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў - рЃўрЃљрЃюрЃЋрЃљрЃарЃў 2026:**
- рЃўрЃгрЃДрЃћрЃЉрЃљ рЃ«рЃБрЃЌрЃерЃљрЃЉрЃљрЃЌрЃА (4) Рєњ 3 рЃфрЃљрЃарЃўрЃћрЃџрЃў
- рЃљрЃЦрЃЋрЃА 31 рЃЊрЃдрЃћ
- рЃЏрЃљрЃАрЃўрЃЋрЃў: `[null, null, null, 1, 2, 3, 4, ..., 31]`

```tsx
const renderDay = () => {
  const days = generateCaledarDays();

  return days.map((day, index) => {
    // рЃфрЃљрЃарЃўрЃћрЃџрЃў рЃБрЃ»рЃарЃљ
    if (day === null) {
      return <div key={index} className={styles.emptyDay}></div>;
    }
    
    // рЃЊрЃдрЃћрЃЋрЃљрЃюрЃЊрЃћрЃџрЃў?
    const today = new Date();
    const isToday =
      today.getDate() === day &&
      today.getMonth() === currentMonth.getMonth() &&
      today.getFullYear() === currentMonth.getFullYear();

    // рЃљрЃарЃЕрЃћрЃБрЃџрЃў?
    const isSelected =
      selectedDate &&
      selectedDate.getDate() === day &&
      selectedDate.getMonth() === currentMonth.getMonth() &&
      selectedDate.getFullYear() === currentMonth.getFullYear();

      <div
        key={index}

  });
```
